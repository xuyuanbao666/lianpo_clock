# 廉颇时钟 - 番茄时钟应用实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个Android原生番茄时钟应用，支持自定义时间、任务管理、统计报告等功能，采用Jetpack Compose和MVVM架构。

**Architecture:** 使用MVVM架构，Jetpack Compose构建UI，ViewModel管理状态，Room数据库存储数据，WorkManager处理后台计时和通知。采用单Activity多Composable的导航结构。

**Tech Stack:** Kotlin, Jetpack Compose, Room, WorkManager, Hilt, Material3

package com.lianpo.clock.ui.theme

import android.app.Activity
import android.os.Build
import androidx.compose.foundation.isSystemInDarkTheme
import androidx.compose.material3.MaterialTheme
import androidx.compose.material3.darkColorScheme
import androidx.compose.material3.dynamicDarkColorScheme
import androidx.compose.material3.dynamicLightColorScheme

import androidx.compose.runtime.SideEffect
import androidx.compose.ui.graphics.toArgb
import androidx.compose.ui.platform.LocalContext
import androidx.compose.ui.platform.LocalView
import androidx.core.view.WindowCompat

private val DarkColorScheme = darkColorScheme(
    primary = Primary,
    onPrimary = OnPrimary,
    primaryContainer = PrimaryContainer,
    onPrimaryContainer = OnPrimaryContainer,
    secondary = Secondary,
    onSecondary = OnSecondary,
    secondaryContainer = SecondaryContainer,
    onSecondaryContainer = OnSecondaryContainer,
    tertiary = Tertiary,
    onTertiary = OnTertiary,
    tertiaryContainer = TertiaryContainer,
    onTertiaryContainer = OnTertiaryContainer,
    background = Background,
    onBackground = OnBackground,
    surface = Surface,
    onSurface = OnSurface,
    error = Error,
    onError = OnError,
    errorContainer = ErrorContainer,
    onErrorContainer = OnErrorContainer
)

private val LightColorScheme = lightColorScheme(
    primary = Primary,
    onPrimary = OnPrimary,
    primaryContainer = PrimaryContainer,
    onPrimaryContainer = OnPrimaryContainer,
    secondary = Secondary,
    onSecondary = OnSecondary,
    secondaryContainer = SecondaryContainer,
    onSecondaryContainer = OnSecondaryContainer,
    tertiary = Tertiary,
    onTertiary = OnTertiary,
    tertiaryContainer = TertiaryContainer,
    onTertiaryContainer = OnTertiaryContainer,
    background = Background,
    onBackground = OnBackground,
    surface = Surface,
    onSurface = OnSurface,
    error = Error,
    onError = OnError,
    errorContainer = ErrorContainer,
    onErrorContainer = OnErrorContainer
)

@Composable
fun LianpoClockTheme(
    darkTheme: Boolean = isSystemInDarkTheme(),
    dynamicColor: Boolean = true,
    content: @Composable () -> Unit
) {
    val colorScheme = when {
        dynamicColor && Build.VERSION.SDK_INT >= Build.VERSION_CODES.S -> {
            val context = LocalContext.current
            if (darkTheme) dynamicDarkColorScheme(context) else dynamicLightColorScheme(context)
        }
        darkTheme -> DarkColorScheme
        else -> LightColorScheme
    }
    
    val view = LocalView.current
    if (!view.isInEditMode) {
        SideEffect {
            val window = (view.context as Activity).window
            window.statusBarColor = colorScheme.primary.toArgb()
            WindowCompat.getInsetsController(window, view).isAppearanceLightStatusBars = !darkTheme
        }
    }

    MaterialTheme(
        colorScheme = colorScheme,
        typography = Typography,
        content = content
    )
}

package com.lianpo.clock.data.database.entity

import androidx.room.Entity
import androidx.room.ForeignKey
import androidx.room.Index
import androidx.room.PrimaryKey

@Entity(
    tableName = "pomodoro_records",
    foreignKeys = [
        ForeignKey(
            entity = Task::class,
            parentColumns = ["id"],
            childColumns = ["taskId"],
            onDelete = ForeignKey.SET_NULL
        )
    ],
    indices = [Index(value = ["taskId"])]
)
data class PomodoroRecord(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    val taskId: Long? = null, // 关联的任务ID，可以为空
    val startTime: Long,
    val endTime: Long,
    val duration: Int, // 持续时间（分钟）
    val isCompleted: Boolean = true, // 是否完成（未被中断）
    val type: PomodoroType = PomodoroType.WORK
)

enum class PomodoroType {
    WORK, // 工作时间
    SHORT_BREAK, // 短休息
    LONG_BREAK // 长休息
}

package com.lianpo.clock.data.database.dao

import androidx.room.*
import com.lianpo.clock.data.database.entity.Task
import kotlinx.coroutines.flow.Flow

@Dao
interface TaskDao {
    @Query("SELECT * FROM tasks ORDER BY createdAt DESC")
    fun getAllTasks(): Flow<List<Task>>
    
    @Query("SELECT * FROM tasks WHERE id = :taskId")
    suspend fun getTaskById(taskId: Long): Task?
    
    @Query("SELECT * FROM tasks WHERE isCompleted = 0 ORDER BY createdAt DESC")
    fun getActiveTasks(): Flow<List<Task>>
    
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertTask(task: Task): Long
    
    @Update
    suspend fun updateTask(task: Task)
    
    @Delete
    suspend fun deleteTask(task: Task)
    
    @Query("DELETE FROM tasks WHERE id = :taskId")
    suspend fun deleteTaskById(taskId: Long)
    
    @Query("UPDATE tasks SET isCompleted = 1, completedAt = :completedAt WHERE id = :taskId")
    suspend fun markTaskAsCompleted(taskId: Long, completedAt: Long = System.currentTimeMillis())
    
    @Query("UPDATE tasks SET pomodoroCount = pomodoroCount + 1 WHERE id = :taskId")
    suspend fun incrementPomodoroCount(taskId: Long)
}

package com.lianpo.clock.data.database.dao

import androidx.room.*
import com.lianpo.clock.data.database.entity.PomodoroRecord
import com.lianpo.clock.data.database.entity.PomodoroType
import kotlinx.coroutines.flow.Flow

@Dao
interface PomodoroDao {
    @Query("SELECT * FROM pomodoro_records ORDER BY startTime DESC")
    fun getAllRecords(): Flow<List<PomodoroRecord>>
    
    @Query("SELECT * FROM pomodoro_records WHERE id = :recordId")
    suspend fun getRecordById(recordId: Long): PomodoroRecord?
    
    @Query("SELECT * FROM pomodoro_records WHERE taskId = :taskId ORDER BY startTime DESC")
    fun getRecordsByTaskId(taskId: Long): Flow<List<PomodoroRecord>>
    
    @Query("SELECT * FROM pomodoro_records WHERE startTime >= :startTime AND startTime <= :endTime ORDER BY startTime DESC")
    fun getRecordsBetween(startTime: Long, endTime: Long): Flow<List<PomodoroRecord>>
    
    @Query("SELECT * FROM pomodoro_records WHERE type = :type ORDER BY startTime DESC")
    fun getRecordsByType(type: PomodoroType): Flow<List<PomodoroRecord>>
    
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertRecord(record: PomodoroRecord): Long
    
    @Update
    suspend fun updateRecord(record: PomodoroRecord)
    
    @Delete
    suspend fun deleteRecord(record: PomodoroRecord)
    
    @Query("SELECT SUM(duration) FROM pomodoro_records WHERE startTime >= :startTime AND startTime <= :endTime AND type = 'WORK'")
    suspend fun getTotalWorkDurationBetween(startTime: Long, endTime: Long): Int?
    
    @Query("SELECT COUNT(*) FROM pomodoro_records WHERE startTime >= :startTime AND startTime <= :endTime AND type = 'WORK'")

// app/src/main/java/com/lianpo/clock/ui/timer/TimerViewModel.kt
package com.lianpo.clock.ui.timer

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.lianpo.clock.data.database.entity.PomodoroRecord
import com.lianpo.clock.data.database.entity.PomodoroType
import com.lianpo.clock.data.repository.PomodoroRepository
import com.lianpo.clock.data.repository.TaskRepository
import com.lianpo.clock.util.TimerConfig
import com.lianpo.clock.util.TimerState
import com.lianpo.clock.util.TimerType
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.Job
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch
import javax.inject.Inject

@HiltViewModel
class TimerViewModel @Inject constructor(
    private val pomodoroRepository: PomodoroRepository,
    private val taskRepository: TaskRepository
) : ViewModel() {
    
    private val _timerState = MutableStateFlow(TimerState.IDLE)
    val timerState: StateFlow<TimerState> = _timerState.asStateFlow()
    
    private val _timerType = MutableStateFlow(TimerType.WORK)
    val timerType: StateFlow<TimerType> = _timerType.asStateFlow()
    
    private val _timeRemaining = MutableStateFlow(0) // 秒
    val timeRemaining: StateFlow<Int> = _timeRemaining.asStateFlow()
    
    private val _totalTime = MutableStateFlow(0) // 秒
    val totalTime: StateFlow<Int> = _totalTime.asStateFlow()
    
    private val _completedPomodoros = MutableStateFlow(0)
    val completedPomodoros: StateFlow<Int> = _completedPomodoros.asStateFlow()
    
    private val _currentTaskId = MutableStateFlow<Long?>(null)
    val currentTaskId: StateFlow<Long?> = _currentTaskId.asStateFlow()
    
    private val timerConfig = TimerConfig()
    private var timerJob: Job? = null
    private var startTime: Long = 0
    
    init {
        resetTimer()
    }
    
    fun startTimer() {
        if (_timerState.value == TimerState.IDLE || _timerState.value == TimerState.PAUSED) {
            _timerState.value = TimerState.RUNNING
            startTime = System.currentTimeMillis()
            startCountdown()
        }
    }
    
    fun pauseTimer() {
        if (_timerState.value == TimerState.RUNNING) {
            _timerState.value = TimerState.PAUSED
            timerJob?.cancel()
        }
    }
    
    fun resetTimer() {
        timerJob?.cancel()
        _timerState.value = TimerState.IDLE
        _timerType.value = TimerType.WORK
        _timeRemaining.value = timerConfig.workDuration * 60
        _totalTime.value = timerConfig.workDuration * 60
    }
    
    fun skipToNext() {
        timerJob?.cancel()
        _timerState.value = TimerState.IDLE
        moveToNextPhase()
    }
    
    fun setCurrentTask(taskId: Long?) {
        _currentTaskId.value = taskId
    }
    
    private fun startCountdown() {
        timerJob = viewModelScope.launch {
            while (_timeRemaining.value > 0 && _timerState.value == TimerState.RUNNING) {
                delay(1000)
                _timeRemaining.value--
            }
            
            if (_timeRemaining.value <= 0) {
                _timerState.value = TimerState.FINISHED
                onTimerFinished()
            }
        }
    }
    
    private fun onTimerFinished() {
        viewModelScope.launch {
            // 记录完成的番茄钟
            if (_timerType.value == TimerType.WORK) {
                val record = PomodoroRecord(
                    taskId = _currentTaskId.value,
                    startTime = startTime,
                    endTime = System.currentTimeMillis(),
                    duration = timerConfig.workDuration,
                    isCompleted = true,
                    type = PomodoroType.WORK
                )
                pomodoroRepository.insertRecord(record)
                
                // 更新任务的番茄钟计数
                _currentTaskId.value?.let { taskId ->
                    taskRepository.incrementPomodoroCount(taskId)
                }
                
                _completedPomodoros.value++
            }
            
            // 自动移动到下一阶段
            moveToNextPhase()
        }
    }
    
    private fun moveToNextPhase() {
        when (_timerType.value) {
            TimerType.WORK -> {
                if (_completedPomodoros.value % timerConfig.longBreakInterval == 0) {
                    _timerType.value = TimerType.LONG_BREAK
                    _timeRemaining.value = timerConfig.longBreakDuration * 60
                    _totalTime.value = timerConfig.longBreakDuration * 60
                } else {
                    _timerType.value = TimerType.SHORT_BREAK
                    _timeRemaining.value = timerConfig.shortBreakDuration * 60
                    _totalTime.value = timerConfig.shortBreakDuration * 60
                }
            }
            TimerType.SHORT_BREAK, TimerType.LONG_BREAK -> {
                _timerType.value = TimerType.WORK
                _timeRemaining.value = timerConfig.workDuration * 60
                _totalTime.value = timerConfig.workDuration * 60
            }
        }
    }
    
    fun getProgress(): Float {
        if (_totalTime.value == 0) return 0f
        return (_totalTime.value - _timeRemaining.value).toFloat() / _totalTime.value.toFloat()
    }
    
    fun formatTime(seconds: Int): String {
        val minutes = seconds / 60
        val remainingSeconds = seconds % 60
        return "%02d:%02d".format(minutes, remainingSeconds)
    }
}

@Composable
fun CircularProgress(
    progress: Float,
    modifier: Modifier = Modifier,
    size: Dp = 200.dp,
    strokeWidth: Dp = 12.dp,
    backgroundColor: Color = MaterialTheme.colorScheme.surfaceVariant,
    progressColor: Color = MaterialTheme.colorScheme.primary,
    trackColor: Color = Color.Transparent
) {

        targetValue = progress,
        animationSpec = tween(durationMillis = 300),
        label = "progress"
    )

    Box(
        modifier = modifier.size(size)
    ) {
        Canvas(
            modifier = Modifier.size(size)
        ) {
            val canvasSize = this.size.width
            val radius = (canvasSize - strokeWidth.toPx()) / 2
            val center = Offset(canvasSize / 2, canvasSize / 2)
            
            // 绘制背景圆环
            drawCircle(
                color = backgroundColor,
                radius = radius,
                center = center,
                style = Stroke(width = strokeWidth.toPx())
            )
            
            // 绘制进度弧线
            drawArc(
                color = progressColor,
                startAngle = -90f,
                sweepAngle = 360f * animatedProgress.value,
                useCenter = false,
                topLeft = Offset(center.x - radius, center.y - radius),
                size = Size(radius * 2, radius * 2),
                style = Stroke(
                    width = strokeWidth.toPx(),
                    cap = StrokeCap.Round
                )
            )
        }
    }

@HiltViewModel
class StatisticsViewModel @Inject constructor(
    private val pomodoroRepository: PomodoroRepository
) : ViewModel() {
    
    private val _todayStats = MutableStateFlow(DailyStats())
    val todayStats: StateFlow<DailyStats> = _todayStats.asStateFlow()
    
    private val _weeklyStats = MutableStateFlow<List<DailyStats>>(emptyList())
    val weeklyStats: StateFlow<List<DailyStats>> = _weeklyStats.asStateFlow()
    
    private val _totalStats = MutableStateFlow(TotalStats())
    val totalStats: StateFlow<TotalStats> = _totalStats.asStateFlow()
    
    init {
        loadStatistics()
    }
    
    fun loadStatistics() {
        viewModelScope.launch {
            val calendar = Calendar.getInstance()
            
            // 今日统计
            calendar.set(Calendar.HOUR_OF_DAY, 0)
            calendar.set(Calendar.MINUTE, 0)
            calendar.set(Calendar.SECOND, 0)
            calendar.set(Calendar.MILLISECOND, 0)
            val todayStart = calendar.timeInMillis
            
            calendar.add(Calendar.DAY_OF_MONTH, 1)
            val todayEnd = calendar.timeInMillis
            
            val todayDuration = pomodoroRepository.getTotalWorkDurationBetween(todayStart, todayEnd)
            val todayCount = pomodoroRepository.getWorkRecordCountBetween(todayStart, todayEnd)
            
            _todayStats.value = DailyStats(
                date = todayStart,
                focusMinutes = todayDuration,
                pomodoroCount = todayCount
            )
            
            // 本周统计
            calendar.timeInMillis = System.currentTimeMillis()
            calendar.set(Calendar.DAY_OF_WEEK, calendar.firstDayOfWeek)
            calendar.set(Calendar.HOUR_OF_DAY, 0)
            calendar.set(Calendar.MINUTE, 0)
            calendar.set(Calendar.SECOND, 0)
            calendar.set(Calendar.MILLISECOND, 0)
            val weekStart = calendar.timeInMillis
            
            val weeklyStatsList = mutableListOf<DailyStats>()
            for (i in 0 until 7) {
                val dayStart = weekStart + i * 24 * 60 * 60 * 1000
                val dayEnd = dayStart + 24 * 60 * 60 * 1000
                
                val dayDuration = pomodoroRepository.getTotalWorkDurationBetween(dayStart, dayEnd)
                val dayCount = pomodoroRepository.getWorkRecordCountBetween(dayStart, dayEnd)
                
                weeklyStatsList.add(
                    DailyStats(
                        date = dayStart,
                        focusMinutes = dayDuration,
                        pomodoroCount = dayCount
                    )
                )
            }
            
            _weeklyStats.value = weeklyStatsList
            
            // 总计统计
            val allRecords = pomodoroRepository.getAllRecords()
            // 这里需要实现总计统计的逻辑
            _totalStats.value = TotalStats(
                totalFocusMinutes = todayDuration, // 临时使用今日数据
                totalPomodoros = todayCount,
                currentStreak = 1 // 临时数据
            )
        }
    }
}

    val pomodoroCount: Int = 0

import androidx.compose.ui.unit.dp
import com.lianpo.clock.ui.statistics.DailyStats
import java.text.SimpleDateFormat
import java.util.*

@Composable
fun DailyChart(
    stats: List<DailyStats>,
    modifier: Modifier = Modifier
) {
    if (stats.isEmpty()) return
    
    val maxValue = stats.maxOf { it.focusMinutes }.toFloat()
    val barColor = MaterialTheme.colorScheme.primary
    val textColor = MaterialTheme.colorScheme.onSurface
    
    Column(
        modifier = modifier.fillMaxWidth()
    ) {
        Text(
            text = "本周专注时间（分钟）",
            style = MaterialTheme.typography.titleMedium,
            modifier = Modifier.padding(bottom = 8.dp)
        )
        
        Canvas(
            modifier = Modifier
                .fillMaxWidth()
                .height(200.dp)
        ) {
            val canvasWidth = size.width
            val canvasHeight = size.height
            val barWidth = canvasWidth / stats.size * 0.6f
            val barSpacing = canvasWidth / stats.size * 0.4f
            
            stats.forEachIndexed { index, dailyStats ->
                val barHeight = if (maxValue > 0) {
                    (dailyStats.focusMinutes / maxValue) * canvasHeight * 0.8f
                } else {
                    0f
                }
                
                val x = index * (barWidth + barSpacing) + barSpacing / 2
                val y = canvasHeight - barHeight
                
                // 绘制柱状图
                drawRect(
                    color = barColor,
                    topLeft = Offset(x, y),
                    size = Size(barWidth, barHeight)
                )
                
                // 绘制日期标签
                val dateFormat = SimpleDateFormat("MM/dd", Locale.getDefault())
                val dateStr = dateFormat.format(Date(dailyStats.date))
                
                drawContext.canvas.nativeCanvas.apply {
                    val paint = android.graphics.Paint().apply {
                        color = textColor.hashCode()
                        textSize = 24f
                        textAlign = android.graphics.Paint.Align.CENTER
                    }
                    drawText(
                        dateStr,
                        x + barWidth / 2,
                        canvasHeight + 30f,
                        paint
                    )
                }
            }
        }
    }

            // 绘制折线

                color = lineColor,

package com.lianpo.clock.ui.settings

import android.content.Context
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope

import dagger.hilt.android.lifecycle.HiltViewModel
import dagger.hilt.android.qualifiers.ApplicationContext

import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch
import javax.inject.Inject

@HiltViewModel
class SettingsViewModel @Inject constructor(
    @ApplicationContext private val context: Context
) : ViewModel() {
    
    private val _timerConfig = MutableStateFlow(TimerConfig())
    val timerConfig: StateFlow<TimerConfig> = _timerConfig.asStateFlow()
    
    private val _selectedSound = MutableStateFlow("default")
    val selectedSound: StateFlow<String> = _selectedSound.asStateFlow()
    
    private val _notificationsEnabled = MutableStateFlow(true)
    val notificationsEnabled: StateFlow<Boolean> = _notificationsEnabled.asStateFlow()
    
    init {
        loadSettings()
    }
    
    private fun loadSettings() {
        val prefs = context.getSharedPreferences("settings", Context.MODE_PRIVATE)
        
        _timerConfig.value = TimerConfig(
            workDuration = prefs.getInt("work_duration", 25),
            shortBreakDuration = prefs.getInt("short_break_duration", 5),
            longBreakDuration = prefs.getInt("long_break_duration", 15),
            longBreakInterval = prefs.getInt("long_break_interval", 4)
        )
        
        _selectedSound.value = prefs.getString("selected_sound", "default") ?: "default"
        _notificationsEnabled.value = prefs.getBoolean("notifications_enabled", true)
    }
    
    fun updateWorkDuration(minutes: Int) {
        _timerConfig.value = _timerConfig.value.copy(workDuration = minutes)
        saveSettings()
    }
    
    fun updateShortBreakDuration(minutes: Int) {
        _timerConfig.value = _timerConfig.value.copy(shortBreakDuration = minutes)
        saveSettings()
    }
    
    fun updateLongBreakDuration(minutes: Int) {
        _timerConfig.value = _timerConfig.value.copy(longBreakDuration = minutes)
        saveSettings()
    }
    
    fun updateLongBreakInterval(interval: Int) {
        _timerConfig.value = _timerConfig.value.copy(longBreakInterval = interval)
        saveSettings()
    }
    
    fun updateSelectedSound(sound: String) {
        _selectedSound.value = sound
        saveSettings()
    }
    
    fun updateNotificationsEnabled(enabled: Boolean) {
        _notificationsEnabled.value = enabled
        saveSettings()
    }
    
    private fun saveSettings() {
        viewModelScope.launch {
            val prefs = context.getSharedPreferences("settings", Context.MODE_PRIVATE)
            prefs.edit().apply {
                putInt("work_duration", _timerConfig.value.workDuration)
                putInt("short_break_duration", _timerConfig.value.shortBreakDuration)
                putInt("long_break_duration", _timerConfig.value.longBreakDuration)
                putInt("long_break_interval", _timerConfig.value.longBreakInterval)
                putString("selected_sound", _selectedSound.value)
                putBoolean("notifications_enabled", _notificationsEnabled.value)
                apply()
            }
        }
    }
}

package com.lianpo.clock.util

import android.content.Context
import android.media.MediaPlayer
import com.lianpo.clock.R

class SoundPlayer(private val context: Context) {
    
    private var mediaPlayer: MediaPlayer? = null
    
    fun playSound(soundType: String) {
        stopSound()
        
        val soundResource = when (soundType) {
            "crisp" -> R.raw.sound_crisp
            "gentle" -> R.raw.sound_gentle
            else -> R.raw.sound_default
        }
        
        mediaPlayer = MediaPlayer.create(context, soundResource).apply {
            setOnCompletionListener {
                stopSound()
            }
            start()
        }
    }
    
    fun stopSound() {
        mediaPlayer?.apply {
            if (isPlaying) {
                stop()
            }
            release()
        }
        mediaPlayer = null
    }
}
